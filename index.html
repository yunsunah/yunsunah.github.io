Slider API

- 기능 옵션

let defaults = {
	paging: false, //슬라이드 페이징 노출
	control: true, //슬라이드 이전, 다음버튼 노출
	speed: 1000, //이전, 다음버튼 눌렀을 때 슬라이드 이동속도
	autoplay: false, //슬라이드 자동롤링
	playdirection: true, //슬라이드 방향 true:오른쪽, false:왼쪽
	autospeed: 3000, //슬라이드 자동롤링 속도
	playControl: false //정지,재생버튼 노출
};

- 태그 네임 변경

마크업시 슬라이더 기본 태그 클래스명이 변경되거나 id값으로 변경하는 경우 옵션으로 슬라이더 클래스명을 변경해 줄 수 있습니다.

let defaults = {
	frame: ".slide_frame", //슬라이드 감싸는 클래스
	list: ".slide_list", //슬라이드 리스트 클래스
	slide: ".slide_bx", //슬라이드 아이템 클래스
	pagingEl: ".slide_paging", //슬라이드 페이징 클래스
	controlEl: ".slide_control",//슬라이드 이전, 다음 버튼 감싸는 클래스
	prevBtn: ".btn_preview", //슬라이드 이전버튼 클래스
	nextBtn: ".btn_next", //슬라이드 다음버튼 클래스
	playControlEl: ".play_control", //슬라이드 정지,재생 감싸는 클래스
	btnStop: ".btn_stop", //슬라이드 정지버튼 클래스
	btnPlay: ".btn_play", //슬라이드 다음버튼 클래스
};
